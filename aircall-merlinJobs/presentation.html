  • Webhook Code
    •                 // Require express, body-parser, mySQL
const express = require("express");
const bodyParser = require("body-parser");
const mysql = require('mysql');
// Initialize express and define a port
const app = express();
const PORT = 3000;
// Database connection
const con = mysql.createConnection({
  host: "localhost",
  user: "yourusername",
  password: "yourpassword",
  database: "mydb"
});
// Tell express to use body-parser's JSON parsing
app.use(bodyParser.json());
// Start express on the defined port
app.listen(PORT, () => console.log(`🚀 Server running on port ${PORT}`));
// function to parse webhook JSON
function parseWebhookJSON(requestBody) {
  // retrieve data
  const callId = requestBody.data.id,
  type = requestBody.data.direction,
  event = requestBody.event,
  userId = requestBody.data.user.id ?? '',
  dateTime = new Date(requestBody.data.started_at * 1000).toLocaleString();
  // insert into database
  con.connect(function(err) {
    if (err) throw err;
    var sql = "INSERT INTO calls (id,type,event,user_id,date_time) VALUES (callId,type,event,userId,dateTime)";
    con.query(sql, function (err, result) {
      if (err) throw err;
    });
  });
  return true
};
// Handle incoming JSON request
app.use(bodyParser.json());
app.post("/hook", (req, res) => {
  try {
    const response = parseWebhookJSON(req);
    if(response !== true) {
      throw 'issue with JSON parsing';
      res.status(200).end()
    } else res.status(200).end();
  } catch (err) {
    console.error('Error in aircall Webhook: ', err);
  }
});
